#include <bits/stdc++.h>
using namespace std;

    void generateCombos(int idx, int k, int n, vector<int>& temp, vector<vector<int>>& sol){
        if(!k||!n){
            if(!k&&!n) sol.push_back(temp);
            return;
        }

        for(int i=idx;i<=9;i++){
            if(n<i) return;

            temp.push_back(i);
            generateCombos(i+1, k-1, n-i, temp, sol);
            temp.pop_back();
        }
    }

    vector<vector<int>> combinationSum(int k, int n) {
        vector<int> temp;
        vector<vector<int>> sol;

        generateCombos(0, k, n, temp, sol);
        return sol;
    }

int main() {
int n,sum;
cout<<"Enter the n: ";
cin>>n;
cout<<"Enter the target sum: ";
cin>>sum;
vector<vector<int>> v;
v = combinationSum(n,sum);
int count = 0;
for (int i = 0; i < v.size(); i++) {
for (int j = 0; j < v[i].size(); j++)
cout << v[i][j] << " ";
count++;
cout << endl;
}
cout<<"No of combinations is: "<<count<<endl;
return 0;
}
