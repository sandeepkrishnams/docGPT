#include <iostream>
using namespace std;
int main() {
    int arr[] = { 64, 34, 25, 12, 22, 11, 90 };
    int length = sizeof(arr)/sizeof(arr[0]);
    
    for(int i = 0; i < length-1 ; i++) {
        for(int j = 0; j < length-i-1; j++) {
            if(arr[j+1]>arr[j]) {
                int temp;
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    for (int i = 0; i < length; i++) {
        cout << arr[i] << " ";
    }
}
